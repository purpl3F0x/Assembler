26     auto it = reg.find(s);
27     if (it!=reg.end()) return it->second;
28     else throw std::logic_error(s + " is not a  register");

52   std::string tobinary(int n, int d = 16) {
53     std::string s;
54     while (d--) {
55       s = std::to_string(n%2) + s;
56       n = n >> 1;
57     } 
58     return s;
59   }

61   bool Assembler::translate(std::string outName) {
62     //get file name 
63     if (outName=="")
64       outName = 
65           cur_file.substr(0, cur_file.find_last_of("."))
66               + ".bin";
67    
68     if (onDebug) cout << "Opening " << outName << " for writing ...\n";
69    
70     //open file for writing 
71     ofstream outfile; 
72     outfile.open(outName, ios::binary | ios::out);
73    
74     if (!outfile) {
75       cout << "Cannot Open File\n\n";
76       return false;
77     } 
78    
79     //Set All Vector values to zero
80     for (int i = 0; i < SIZE; i++)
81       rom.push_back(0);
82    
83     int romAddressPointer = 1;
84     int ramAddressPointer = 1;
85    
86     // Initialise memory at runtime
87     for (auto &d : data) {
88       /* Assembles a RI routine for to fill RAM with pre-compliled values 
89        * 
90        * RI 2 args (16-bit) -> 3 ROM addresses -> 3 clk cycles 
91        * */ 
92    
93       int ri = opCodes.find("RI")->binary;
94       rom[romAddressPointer++] = ri << (WORD - OPCODE_SIZE);
95       rom[romAddressPointer++] = ramAddressPointer++;
96       rom[romAddressPointer++] = d.second;
97     } 
98    
99    
100    // Instructions 
101    for (auto inst : instructions) {                                             // parse every instruction
102      int pos = 3;
103      std::vector<std::string>::iterator arg_it = inst.arguments.begin();
104   
105      rom.at(romAddressPointer) = opCodes.getBinary(inst.opCode) << (WORD - OPCODE_SIZE);
106   
107      if (!inst.arguments.size()) continue;                                    // opCodes with no args
108   
109      // Initialise Instructions taking 2 addresses lines
110      if (inst.opCode=="MI" || inst.opCode=="MO" || inst.opCode=="STA" || inst.opCode=="STA") {
111   
112        for (arg_it; arg_it!=inst.arguments.end() - 1; arg_it++)               // loop though gives arguments
113          rom.at(romAddressPointer) += (Register(*arg_it) << pos--*3);
114   
115        rom.at(++romAddressPointer) = Register(*arg_it);
116   
117      } else { 
118        for (auto arg_it : inst.arguments)
119          rom.at(romAddressPointer) += (Register(arg_it) << pos--*3);
120      } 
121      romAddressPointer++;
122    } 
123   
124    for (auto &i : text) {                                                     // add text
125      rom.at(romAddressPointer++) = i.second;
126    } 
127   
128    for (auto adr : rom)                                                       // Write to file
129      outfile.write(reinterpret_cast<const char *>(&adr), sizeof(uint16_t));
130   
131    outfile.close();                                                           // Close file
132   
133   
134    return true;
135  }

159  void Assembler::instruction::add(vector<std::string> &v) {
160    arguments.reserve(arguments.size() + v.size());       //Memory Pre allocation
161    arguments.insert(arguments.end(), v.begin(), v.end());
162    size += v.size();

233  int Assembler::to_int(std::string &s) {
234    if (s.find("0b")==0) {                            // binary
235      return stoi( 
236          std::string(s.begin() + 2, s.end()),
237          nullptr,
238          2 
239      ); 
240    } else if (s.find("0x")==0) {                    // Hexadecimal
241      return stoi( 
242          std::string(s.begin() + 2, s.end()),
243          nullptr,
244          16 
245      ); 
246    } else if (s=="true") return true;
247   
248    else if (s=="false") return false;
249   
250    else if (s[0]=='\'') return s[1];
251   
252    else return std::stoi(s);
253  }

255  bool Assembler::data_typeParser(const std::string &line, map<std::string, short> &stack) {
256   
257    short int_val = 0;
258   
259    //string iterators 
260    auto iter_start = line.begin();
261    auto iter_end = line.end();
262   
263    // Bind 
264    data_type d; 
265    auto setType = [&](auto &ctx) { d.type = _attr(ctx); };
266    auto setName = [&](auto &ctx) { d.name = _attr(ctx); };
267    auto setVal = [&](auto &ctx) { d.value = _attr(ctx); };
268   
269    bool result = parse(
270        iter_start, 
271        iter_end, 
272        skip(rules::comment | space)[rules::name[setType]
273            >> rules::name[setName]
274            >> -rules::value[setVal]]
275            >> *(space | rules::comment)
276    ); 
277   
278    if (result && iter_end!=iter_start) {
279      result = false;
280      errors.emplace_back( 
281          error( 
282              cur_line, 
283              iter_start - line.begin(),
284              "Logic Error",
285              "Unexpected -> " + std::string(iter_start, iter_end),
286              line 
287          ) 
288      ); 
289    } else if (find(rules::types.begin(), rules::types.end(), d.type)!=rules::types.end()) {
290   
291      int_val = to_int(d.value);
292   
293    } else if (opCodes.isOpCode(d.name) 
294        || find(rules::types.begin(), rules::types.end(), d.name)!=rules::types.end()) {
295      errors.emplace_back( 
296          error( 
297              cur_line, 
298              line.find(d.name),
299              "Logic Error",
300              d.name + " is not a valid identifier",
301              line 
302          ) 
303      ); 
304      result = false;
305    } else { 
306      errors.emplace_back( 
307          error( 
308              cur_line, 
309              iter_start - line.begin(),
310              "Syntax Error",
311              d.type + " does not name a type",
312              line 
313          ) 
314      ); 
315      result = false;
316    } 
317   
318    if (onDebug) cout << (boost::format("%s: %s \n\t->%s (%d)\n")%d.type%d.name%d.value%int_val);
319   
320    if (result) stack[d.name] = int_val;
321   
322    return result;
323   
324  }

326  bool Assembler::lineParser(const std::string &line) {
327   
328    //string iterators 
329    auto iter_start = line.begin();
330    auto iter_end = line.end();
331   
332    instruction inst; 
333   
334    // Bind 
335    auto add = [&](auto &ctx) { inst.add(_attr(ctx)); };
336   
337    bool result = parse(
338        iter_start, 
339        iter_end, 
340        skip(rules::comment | space)[rules::name[add]
341            >> *(rules::name[add]%',')]
342            >> *(space | rules::comment)
343    ); 
344   
345    auto *op = opCodes.find(inst.opCode);
346   
347    if (result && inst.size && !op) {    // given identifier is not valid opCode
348      result = false;
349      errors.emplace_back( 
350          error(cur_line, 
351                iter_end - iter_start,
352                "Syntax Error",
353                inst.opCode + " is not a valid opcode.",
354                line 
355          ) 
356      ); 
357    } else if (result && inst.size!=op->numOfArgs) {                 // Check arguments number
358   
359      result = false;
360   
361      errors.emplace_back( 
362          error(cur_line, 
363                iter_start - line.begin(),
364                "Logic Error",
365                inst.opCode + " expects " + to_string(op->numOfArgs) + " argument(s), got " +
366                    to_string(inst.size) + ".",    // Example: ADD expects 3 arguments, got 4
367                line 
368          ) 
369      ); 
370    } 
371   
372    if (iter_end 
373        !=iter_start) {                                               // Create an error if parser hasn't come to end of line
374   
375      result = false;
376   
377      errors.emplace_back( 
378          error(cur_line, 
379                iter_start - line.begin(),
380                "Syntax Error",
381                "Unexpected -> " + std::string(iter_start, iter_end),
382                line 
383          ) 
384      ); 
385    } 
386   
387    // Code used for debugging
388    if (onDebug) { 
389      cout << (boost::format("%d: %4s (%d)\n")%cur_line%op->name%op->numOfArgs);
390   
391      for (auto i : inst.arguments)
392        cout << "\t->" << i << "\n";
393    } 
394   
395    // Add instruction to stack
396    if (result) instructions.push_back(inst);
397   
398    return result;
399  }

414  bool Assembler::parser(ifstream &fs) {
415   
416    bool parseData = false;
417    bool parseText = false;
418    bool parseStart = false;
419   
420    cur_line = 0;
421    std::string line;
422    bool success = true;
423   
424    while (std::getline(fs, line)) {
425   
426      cur_line++;
427   
428      if (isEmptyLine(line)) continue;          //skip empty lines
429   
430      if (isData(line)) { 
431        if (parseData) {                        // Return Error if Already have parsed Data section
432          errors.emplace_back( 
433              error(cur_line,
434                    line.find('s'),             // Find where section starts
435                    "Logic Error",
436                    "Redefinition of .data ,already defined at " + std::to_string(parseData),
437                    line 
438              ) 
439          ); 
440          success = false;
441        } else parseData = true;
442        continue;
443      } 
444   
445      if (isText(line)) { 
446        if (parseText) {                           // Return Error if Already have parsed Text section
447          errors.emplace_back( 
448              error(cur_line,
449                    line.find('s'),                // Find where section starts
450                    "Logic Error",
451                    "Redefinition of .text ,already defined at " + std::to_string(parseData),
452                    line 
453              ) 
454          ); 
455          success = false;
456        } else parseText = true;
457        continue;
458      } 
459   
460      if (isStart(line)) { 
461        if (parseStart) {                           // Return Error if Already have parsed Start section
462          errors.emplace_back( 
463              error(cur_line,
464                    line.find('s'),                 // Find where section starts
465                    "Logic Error",
466                    "Redefinition of .text ,already defined at " + std::to_string(parseData),
467                    line 
468              ) 
469          ); 
470          success = false;
471        } else parseStart = true;
472        continue;
473      } 
474   
475      if (parseStart) 
476        success = lineParser(line) && success;
477   
478      else if (parseText) 
479        success = data_typeParser(line, text) && success;
480   
481      else if (parseData) 
482        success = data_typeParser(line, data) && success;
483   
484      else { 
485        errors.emplace_back(                                //throw error for instructions/declerations outside of sections
486            error(cur_line,
487                  line.find('s'),
488                  "Syntax Error",
489                  "Unexpeced outside of section",
490                  line 
491            ) 
492        ); 
493        success = false;
494      } 
495   
496    } //end of parse loop 
497   
498   
499    if (success) text[".start"] = data.size();             // Move .start to pointer set memory values on start
500   
501    if (!success && onDebug)
502      for (auto e : errors)
503        cout << e.get() << "\n";
504   
505    if (onDebug) {                                          // Print Variables values onDebug mode
506      cout << "_______" << " data " << "_______" << "\n";
507      for (auto &i:data)
508        cout << i.first << " : " << i.second << "\n";
509   
510      cout << "_______" << " text " << "_______" << "\n";
511      for (auto &i:text)
512        cout << i.first << " : " << i.second << "\n";
513    } 
514   
515    return success;
516  }
