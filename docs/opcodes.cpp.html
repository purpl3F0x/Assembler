13   unsigned int OpCodes::getBinary(std::string name) const {
14     for (auto op: *this)
15       if (op.name==name)
16         return op.binary;
17    
18     return 0; // returns NOP
19   } 
20    
21   unsigned int OpCodes::numOfArgs(std::string name) const {
22     for (auto op: *this)
23       if (op.name==name)
24         return op.numOfArgs;
25    
26     return 0;
27   } 
28    
29   std::string OpCodes::getName(unsigned short bin) const {
30     for (auto op: *this)
31       if (op.binary==bin)
32         return op.name;
33    
34     return "";
35   } 
36    
37   bool OpCodes::isOpCode(std::string name) const {
38     for (auto op: *this)
39       if (op.name==name)
40         return true;
41    
42     return false;
43   } 
44    
45   opCode *OpCodes::find(std::string name) {
46     for (auto &&op: *this)
47       if (op.name==name)
48         return &op;
49    
50     return nullptr;
51   } 
52    
53   std::string OpCodes::operator[](int bin) {
54     return this->getName(bin);
55   } 
56    
57   int OpCodes::operator[](std::string name) {
58     return this->getBinary(name);
59   }
