namespace assembler{ 
    namespace x3 = boost::spirit::x3; 
    namespace ascii = boost::spirit::x3::ascii; 
 
    using x3::double_; 
    using x3::phrase_parse; 
    using x3::parse; 
    using x3::lit; 
    using x3::char_; 
    using x3::lexeme; 
    using x3::alpha; 
    using x3::alnum; 
    using x3::rule; 
    using ascii::space; 
 
    auto name = x3::rule<class name>{} 
                        =  alpha >> *alnum; 
 
    auto l = x3::rule<class l>{} 
                     = name % ','; 
 
    bool is_instruction(std::string s){ 
        auto iter_start = s.begin(); 
        auto iter_end = s.end(); 
 
        bool result = parse(iter_start,iter_end,l); 
 
 
    } 
 
} 
 
 
int main(){ 
 
    std::string s = "ADD % add,ass"; 
 
    auto iter_start = s.begin(); 
    auto iter_end = s.end(); 
 
    auto name = x3::rule<class name>{} 
            = *char_(' ') >> alpha >> *alnum >> *char_(' ') ; 
 
    auto l = x3::rule<class l>{} 
            = name >> *( ','>> *space ); 
 
 
    bool result = parse( 
            iter_start, 
            iter_end, 
            name >> -("%" >> l) 
            ) && iter_end==iter_start; 
 
 
 
    std::cout<<std::boolalpha<<result<<std::endl; 
 
    if (iter_end != iter_start){ 
        std::cout  << "Unexpected:\n" 
                   <<std::string(iter_start,iter_end); 
 
    } 
 
 
}
